🏗️ Object-Oriented Programming C++
▼
Class Definition
class Animal {
private:
    string name;

protected:
    int age;

public:
    // Constructor with initialiser list
    Animal(string n, int a) : name(n), age(a) {}

    // Copy constructor
    Animal(const Animal& other) : name(other.name), age(other.age) {}

    // Destructor
    virtual ~Animal() {}

    // Getters / setters
    string getName() const { return name; }
    void setAge(int a) { age = a; }

    // Virtual method (can be overridden)
    virtual void speak() const {
        cout << name << " makes a sound\n";
    }
};

class Dog : public Animal {
public:
    Dog(string n) : Animal(n, 0) {}
    void speak() const override {    // override keyword
        cout << getName() << ": Woof!\n";
    }
};

🧠 Memory Management C++
▼
new / delete & Smart Pointers
// Raw (manual — avoid in modern C++)
int* p = new int(42);
int* arr = new int[10];
delete p;          // single object
delete[] arr;      // array

// Smart pointers — prefer these
#include <memory>

// unique_ptr: sole ownership
auto up = make_unique<Dog>("Rex");
up->speak();
// auto-freed when up goes out of scope

// shared_ptr: shared ownership
auto sp1 = make_shared<Dog>("Max");
auto sp2 = sp1;    // ref count = 2
sp1.use_count();   // 2

// weak_ptr: non-owning reference
weak_ptr<Dog> wp = sp1;
if (auto locked = wp.lock()) {
    locked->speak();
}
Move Semantics
// Move constructor
class Buffer {
    int* data;
    size_t size;
public:
    // Move constructor (steals resources)
    Buffer(Buffer&& other) noexcept
        : data(other.data), size(other.size) {
        other.data = nullptr;  // null out source
        other.size = 0;
    }

    // Move assignment
    Buffer& operator=(Buffer&& other) noexcept {
        if (this != &other) {
            delete[] data;
            data = other.data;
            size = other.size;
            other.data = nullptr;
        }
        return *this;
    }
};

// std::move — cast to rvalue
vector<string> a = {"hello", "world"};
vector<string> b = std::move(a); // a is now empty

Decorators & Closures
from functools import wraps

# Decorator
def timer(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        import time
        start = time.time()
        result = func(*args, **kwargs)
        print(f"{func.__name__}: {time.time()-start:.3f}s")
        return result
    return wrapper

@timer
def slow_fn():
    pass

# Closure
def make_counter(start=0):
    count = [start]
    def increment():
        count[0] += 1
        return count[0]
    return increment

counter = make_counter(10)
counter()  # 11
counter()  # 12

# Generator
def fibonacci():
    a, b = 0, 1
    while True:
        yield a
        a, b = b, a + b

▼
Class Syntax
class Animal:
    species = "Animal"        # class variable

    def __init__(self, name, age):
        self.name = name      # instance variable
        self._age = age       # convention: protected
        self.__secret = "x"   # name-mangled

    def __str__(self): return f"{self.name}"
    def __repr__(self): return f"Animal('{self.name}')"
    def __len__(self): return self._age

    @property
    def age(self): return self._age
    @age.setter
    def age(self, v):
        if v < 0: raise ValueError
        self._age = v

    @classmethod
    def create(cls, name): return cls(name, 0)

    @staticmethod
    def is_alive(): return True

class Dog(Animal):
    def __init__(self, name):
        super().__init__(name, 0)

    def speak(self): return f"{self.name}: Woof!"

Error Handling
try:
    result = 10 / 0
except ZeroDivisionError as e:
    print(f"Error: {e}")
except (TypeError, ValueError) as e:
    print(f"Type/Value error: {e}")
except Exception as e:
    print(f"Unexpected: {e}")
else:
    print("No error occurred")
finally:
    print("Always runs")

# Raise exception
def validate(age):
    if age < 0:
        raise ValueError(f"Invalid age: {age}")

# Custom exception
class AppError(Exception):
    def __init__(self, msg, code=0):
        super().__init__(msg)
        self.code = code

# Context manager
class ManagedFile:
    def __init__(self, path): self.path = path
    def __enter__(self):
        self.file = open(self.path)
        return self.file
    def __exit__(self, *args):
        self.file.close()
